# 49. Temperature Converter GUI 
```
from tabulate import tabulate

class Converter:

    def celsius_to_fahrenheit(self, c):
        return (c * 9 / 5) + 32

    def fahrenheit_to_celsius(self, f):
        return (f - 32) * 5 / 9

    def celsius_to_kelvin(self, c):
        return c + 273.15

    def kelvin_to_celsius(self, k):
        return k - 273.15


converter = Converter()

print("1. Celsius to Fahrenheit")
print("2. Fahrenheit to Celsius")
print("3. Celsius to Kelvin")
print("4. Kelvin to Celsius")

choice = int(input("Enter your choice: "))
temp = float(input("Enter temperature: "))

result = 0
conversion = ""

if choice == 1:
    result = converter.celsius_to_fahrenheit(temp)
    conversion = "Celsius → Fahrenheit"

elif choice == 2:
    result = converter.fahrenheit_to_celsius(temp)
    conversion = "Fahrenheit → Celsius"

elif choice == 3:
    result = converter.celsius_to_kelvin(temp)
    conversion = "Celsius → Kelvin"

elif choice == 4:
    result = converter.kelvin_to_celsius(temp)
    conversion = "Kelvin → Celsius"

else:
    print("Invalid Choice")
    exit()

table = [[conversion, f"{result:.2f}"]]

print(tabulate(table,
               headers=["Conversion", "Result"],
               tablefmt="grid"))

# 52. Gradebook System 
```
import json

students = {}


def add_grade(student, grade):

    if grade < 0 or grade > 100:
        print("Invalid Grade")
        return

    if student not in students:
        students[student] = []

    students[student].append(grade)


def calculate_gpa(student):

    if student not in students or len(students[student]) == 0:
        return 0

    return sum(students[student]) / len(students[student])


def save_data():

    with open("grades.json", "w") as file:
        json.dump(students, file)

    print("Data Saved")


def load_data():

    global students

    try:
        with open("grades.json", "r") as file:
            students = json.load(file)

        print("Data Loaded")

    except FileNotFoundError:
        print("File Not Found")


def class_average():

    total = 0
    count = 0

    for grades in students.values():
        total += sum(grades)
        count += len(grades)

    if count == 0:
        return 0

    return total / count


add_grade("John", 85)
add_grade("John", 90)
add_grade("Mary", 75)
add_grade("Mary", 80)

print("John GPA:", calculate_gpa("John"))
print("Mary GPA:", calculate_gpa("Mary"))

save_data()
load_data()

print("Class Average:", class_average())
```

# 53. Task Scheduler 
```
from datetime import datetime

class Task:

    def __init__(self, name, due_date, priority):
        self.name = name
        self.due_date = datetime.strptime(due_date, "%Y-%m-%d")
        self.priority = priority

    def __str__(self):
        return f"{self.name} | Due: {self.due_date.date()} | Priority: {self.priority}"


tasks = [
    Task("Project Report", "2026-06-10", 1),
    Task("Assignment", "2026-05-20", 2),
    Task("Presentation", "2026-06-05", 1),
    Task("Lab Record", "2026-05-15", 3)
]

tasks.sort(key=lambda task: task.due_date)

print("Task Schedule")

for task in tasks:
    print(task)

today = datetime.now()

overdue_tasks = [task for task in tasks if task.due_date < today]

print("\nOverdue Tasks")

for task in overdue_tasks:
    print(task)

# 54. Inventory Manager 
```
class Product:

    def __init__(self, name, stock):
        self.name = name
        self.stock = stock


class Perishable(Product):

    def __init__(self, name, stock):
        super().__init__(name, stock)


class Electronics(Product):

    def __init__(self, name, stock):
        super().__init__(name, stock)


inventory = {
    "Milk": Perishable("Milk", 5),
    "Bread": Perishable("Bread", 12),
    "Laptop": Electronics("Laptop", 3),
    "Mouse": Electronics("Mouse", 20)
}

low_stock = set()

for product in inventory.values():

    if product.stock < 10:
        low_stock.add(product.name)

print("Inventory Summary")

for name, product in inventory.items():
    print(f"{name} : {product.stock} units")

print("\nLow Stock Alerts")

for item in low_stock:
    print(item)
```

# 55. Budget Planner 
```
import matplotlib.pyplot as plt

class Category:

    def __init__(self, name, limit):
        self.name = name
        self.limit = limit
        self.spent = 0

    def add_expense(self, amount):
        self.spent += amount

        if self.spent > self.limit:
            print(f"Alert: {self.name} budget exceeded!")


food = Category("Food", 5000)
travel = Category("Travel", 3000)
shopping = Category("Shopping", 4000)

while True:

    category = input("Enter category (Food/Travel/Shopping) or done: ")

    if category.lower() == "done":
        break

    amount = float(input("Enter amount: "))

    if category == "Food":
        food.add_expense(amount)

    elif category == "Travel":
        travel.add_expense(amount)

    elif category == "Shopping":
        shopping.add_expense(amount)

    else:
        print("Invalid Category")

print("\nBudget Summary")
print("Food:", food.spent)
print("Travel:", travel.spent)
print("Shopping:", shopping.spent)

labels = ["Food", "Travel", "Shopping"]
amounts = [food.spent, travel.spent, shopping.spent]

plt.figure(figsize=(5, 5))
plt.pie(amounts, labels=labels, autopct="%1.1f%%")
plt.title("Monthly Budget Distribution")
plt.show()
```
